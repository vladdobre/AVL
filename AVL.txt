class Node

    Node(int x)

class GfG
{
    public static boolean isBST(Node n, int lower, int upper)
    {
    	if(n==null) return true;
    	if( n.data <= lower || n.data >= upper ) return false;
    	return isBST(n.left, lower, n.data) && isBST(n.right, n.data, upper) ;
    }
    
    public static pair isBalanced(Node n)
    {
    	if(n==null)
    	    {
    	        return new pair(0,true);
    	    }
        
    	pair l = isBalanced(n.left);
    	pair r = isBalanced(n.right);
    
    	if( Math.abs(l.first - r.first) > 1 ) return new pair (0,false);
    
    	return new pair( 1 + Math.max(l.first , r.first) , l.second && r.second );
    }
    
    public static boolean isBalancedBST(Node root)
    {
    	if( isBST(root, Integer.MIN_VALUE , Integer.MAX_VALUE) == false )
    		System.out.print("BST voilated, inorder traversal : ");
    
    	else if ( isBalanced(root).second == false)
    		System.out.print("Unbalanced BST, inorder traversal : ");
    
    	else return true;
    	return false;
    }
    
    public static void printInorder(Node n)
    {
    	if(n==null) return;
    	printInorder(n.left);
    	System.out.print(n.data + " ");
    	printInorder(n.right);
    }

class Solution
{
    
    public  Node insertToAVL(Node node,int data)
    {
        if(node == null) return new Node(data);
        return insertNode(node, data);
        
    }
    
    public Node insertNode(Node node, int data){
        
        int h_r;
        int h_l;
        if(node.data==data){
            return node;
        }
        if(node.data<data){
            if(node.right==null){
                node.right=new Node(data);
                h_r=1;
                h_l=getHeight(node.left);
            }
            else{
                node.right = insertNode(node.right, data);   //here node=4;
                h_r = getHeight(node.right);
                h_l=getHeight(node.left);


            }
        }else{
            if(node.left==null){
                node.left=new Node(data);
                h_l=1;
                h_r=getHeight(node.right);
            }
            else {
                node.left=insertNode(node.left, data);
                h_r = getHeight(node.right);
                h_l=getHeight(node.left);
                
            }
        }
        
        int test_height = h_l-h_r;
        if(Math.abs(test_height)>1){
            if(test_height>1){
                int child_height = getHeight(node.left.left)-getHeight(node.left.right);
                if(child_height>0){
                    //a) y is left child of z and x is left child of y (Left Left Case)
                    node = rightRotate(node);
                }else{
                    node.left = leftRotate(node.left);
                    node = rightRotate(node);
                    //b) y is left child of z and x is right child of y (Left Right Case) 
                }
            }else{
                int child_height = getHeight(node.right.left)-getHeight(node.right.right);
                if(child_height>0){
                    node.right = rightRotate(node.right);   //HERE
                    node = leftRotate(node);
                    //a) y is left child of z and x is left child of y (Left Left Case) 
                }else{
                    node = leftRotate(node);
                    //b) y is left child of z and x is right child of y (Left Right Case) 
                }
            }
        }
        
        return node;
    }
    
    public Node leftRotate(Node node){
        Node temp =node;
        
        node = node.right;
        temp.right = node.left;
        node.left = temp;
        return node;
    }
    
    public Node rightRotate(Node node){
        Node temp = node;
        node = node.left;
        temp.left = node.right;
        node.right=temp;
        return node;
    }
    
    public int getHeight(Node node){
        if(node ==null) return 0;
        int h_r = getHeight(node.right);
        int h_l = getHeight(node.left);
        return 1 + ((h_r>h_l) ? h_r : h_l);
        
    }
}
